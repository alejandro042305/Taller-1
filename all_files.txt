    float y = y0;
    int i = 0;
    while (i < iters) {
        fpu HERON_STEP(x, y);

=== src/enlazador.l ===
%{
/*
 * enlazador.l
 * Enlazador-Cargador NB-64
 * Une multiples archivos .obj y resuelve referencias externas
 * Uso: ./enlazador modulo_a.obj modulo_b.obj -o salida.bin
 */

#include <stdio.h>
#include <stdlib.h>
#include <string.h>

#define MAX_BYTES    4096
#define MAX_SYMBOLS   256
#define MAX_PATCHES   256

/* ── Imagen de memoria final ── */
char imagen[MAX_BYTES][9];
int  total_bytes = 0;

/* ── Tabla de simbolos globales ── */
typedef struct {
    char nombre[64];
    int  addr;
    int  es_global;
} Symbol;

Symbol sym_table[MAX_SYMBOLS];
int    num_symbols = 0;

/* ── Tabla de parches (referencias externas a resolver) ── */
typedef struct {
    char nombre[64];   /* nombre del simbolo externo */
    int  addr_parche;  /* direccion en imagen donde parchear */
    int  tipo;         /* 0=DIR48 en instruccion de 8 bytes  */
} Patch;

Patch patches[MAX_PATCHES];
int   num_patches = 0;

/* ── Funciones ── */
void cargar_obj(const char *filename);
void agregar_simbolo(const char *nombre, int addr, int es_global);
int  buscar_simbolo(const char *nombre);
void agregar_parche(const char *nombre, int addr_parche);
void resolver_parches();
void emitir_imagen(const char *filename);
void parchear_dir48(int addr_parche, int dir);

%}

%option noyywrap

.|\n { }
%%

/* ════════════════════════════════════
   CARGA DE ARCHIVO .OBJ
   ════════════════════════════════════ */

void cargar_obj(const char *filename) {
    FILE *f = fopen(filename, "r");
    if (!f) {
        fprintf(stderr, "[ENLAZADOR] Error: no se puede abrir '%s'\n",
                filename);
        exit(1);
    }

    fprintf(stderr, "[ENLAZADOR] Cargando: %s (base=%d)\n",
            filename, total_bytes);

    int base = total_bytes;
    char linea[256];

    while (fgets(linea, sizeof(linea), f)) {
        /* Quitar \r y \n */
        char *cr = strchr(linea, '\r');
        if (cr) *cr = '\0';
        char *nl = strchr(linea, '\n');
        if (nl) *nl = '\0';

        /* Saltar lineas vacias */
        if (strlen(linea) == 0) continue;

        /* Directiva GLOBAL nombre addr */
        if (strncmp(linea, "GLOBAL ", 7) == 0) {
            char nombre[64];
            int  addr;
            sscanf(linea + 7, "%s %d", nombre, &addr);
            agregar_simbolo(nombre, base + addr, 1);
            fprintf(stderr, "[ENLAZADOR] Simbolo global '%s' → addr %d\n",
                    nombre, base + addr);
            continue;
        }

        /* Directiva EXTERN nombre addr_parche */
        if (strncmp(linea, "EXTERN ", 7) == 0) {
            char nombre[64];
            int  addr_parche;
            sscanf(linea + 7, "%s %d", nombre, &addr_parche);
            agregar_parche(nombre, base + addr_parche);
            fprintf(stderr,
                    "[ENLAZADOR] Parche pendiente '%s' en addr %d\n",
                    nombre, base + addr_parche);
            continue;
        }

        /* Linea binaria de 8 bits */
        if (strlen(linea) == 8) {
            int valido = 1;
            for (int i = 0; i < 8; i++)
                if (linea[i] != '0' && linea[i] != '1')
                    valido = 0;
            if (valido) {
                if (total_bytes >= MAX_BYTES) {
                    fprintf(stderr,
                            "[ENLAZADOR] Error: imagen llena\n");
                    exit(1);
                }
                strncpy(imagen[total_bytes], linea, 8);
                imagen[total_bytes][8] = '\0';
                total_bytes++;
            }
        }
    }

    fclose(f);
    fprintf(stderr, "[ENLAZADOR] '%s' cargado: %d bytes\n",
            filename, total_bytes - base);
}

/* ════════════════════════════════════
   TABLA DE SIMBOLOS
   ════════════════════════════════════ */

void agregar_simbolo(const char *nombre, int addr, int es_global) {
    for (int i = 0; i < num_symbols; i++) {
        if (strcmp(sym_table[i].nombre, nombre) == 0) {
            sym_table[i].addr      = addr;
            sym_table[i].es_global = es_global;
            return;
        }
    }
    strncpy(sym_table[num_symbols].nombre, nombre, 63);
    sym_table[num_symbols].addr      = addr;
    sym_table[num_symbols].es_global = es_global;
    num_symbols++;
}

int buscar_simbolo(const char *nombre) {
    for (int i = 0; i < num_symbols; i++)
        if (strcmp(sym_table[i].nombre, nombre) == 0)
            return sym_table[i].addr;
    return -1;
}

/* ════════════════════════════════════
   TABLA DE PARCHES
   ════════════════════════════════════ */

void agregar_parche(const char *nombre, int addr_parche) {
    strncpy(patches[num_patches].nombre, nombre, 63);
    patches[num_patches].addr_parche = addr_parche;
    patches[num_patches].tipo        = 0;
    num_patches++;
}

/* Escribe una direccion de 48 bits en la imagen
   a partir de addr_parche, respetando el opcode
   en los primeros 7 bits del byte inicial */
void parchear_dir48(int addr_parche, int dir) {
    /* Los 8 bytes de la instruccion DIR:
       [opcode7][dir48][pad9]
       El opcode esta en bits 7-1 del byte 0
       Los bits de dir empiezan en bit 0 del byte 0 */

    /* Construir string de 64 bits de la instruccion completa */
    char instr[65];
    /* Leer los 8 bytes actuales */
    for (int i = 0; i < 8; i++)
        for (int b = 0; b < 8; b++)
            instr[i*8+b] = imagen[addr_parche+i][b];
    instr[64] = '\0';

    /* Reemplazar bits 7..56 con la direccion (48 bits) */
    for (int i = 47; i >= 0; i--) {
        int bit_pos = 7 + (47 - i);
        instr[bit_pos] = ((dir >> i) & 1) ? '1' : '0';
    }

    /* Escribir los 8 bytes de vuelta a la imagen */
    for (int i = 0; i < 8; i++) {
        strncpy(imagen[addr_parche+i], instr + i*8, 8);
        imagen[addr_parche+i][8] = '\0';
    }
}

/* ════════════════════════════════════
   RESOLVER PARCHES
   ════════════════════════════════════ */

void resolver_parches() {
    fprintf(stderr, "[ENLAZADOR] Resolviendo %d parche(s)...\n",
            num_patches);

    for (int i = 0; i < num_patches; i++) {
        int addr = buscar_simbolo(patches[i].nombre);
        if (addr == -1) {
            fprintf(stderr,
                "[ENLAZADOR] Error: simbolo '%s' no resuelto\n",
                patches[i].nombre);
            exit(1);
        }
        fprintf(stderr,
            "[ENLAZADOR] Parcheando '%s': addr %d → pos %d\n",
            patches[i].nombre, addr, patches[i].addr_parche);
        parchear_dir48(patches[i].addr_parche, addr);
    }
}

/* ════════════════════════════════════
   EMITIR IMAGEN FINAL
   ════════════════════════════════════ */

void emitir_imagen(const char *filename) {
    FILE *f = stdout;
    if (filename) {
        f = fopen(filename, "w");
        if (!f) {
            fprintf(stderr,
                "[ENLAZADOR] Error: no se puede crear '%s'\n",
                filename);
            exit(1);
        }
    }

    for (int i = 0; i < total_bytes; i++)
        fprintf(f, "%s\n", imagen[i]);

    if (filename) fclose(f);

    fprintf(stderr,
        "[ENLAZADOR] Imagen final: %d bytes\n", total_bytes);
}

/* ════════════════════════════════════
   MAIN
   ════════════════════════════════════ */

int main(int argc, char *argv[]) {
    if (argc < 2) {
        fprintf(stderr,
            "Uso: %s archivo1.obj [archivo2.obj ...] [-o salida.bin]\n",
            argv[0]);
        return 1;
    }

    char *output_file = NULL;

    /* Procesar argumentos */
    for (int i = 1; i < argc; i++) {
        if (strcmp(argv[i], "-o") == 0) {
            if (i + 1 < argc)
                output_file = argv[++i];
        } else {
            cargar_obj(argv[i]);
        }
    }

    /* Resolver referencias externas */
    resolver_parches();

    /* Emitir imagen final */
    emitir_imagen(output_file);

    return 0;
}

/*

 * Ensamblador NB-64: convierte mnemonics a binario

typedef struct {

Label labels[MAX_LABELS];
int   num_labels  = 0;
int   current_addr = 0;
int   pasada       = 1;

/* ── Buffer de salida ── */

int  num_output = 0;

/* ── Funciones auxiliares ── */

/* ════════════════════════════════════════
   UTILIDADES DE CODIFICACIÓN
   ════════════════════════════════════════ */

/* Convierte entero a binario de 8 bits */
void int_to_bin8(int v, char *out) {
    for (int i = 7; i >= 0; i--)
        out[7-i] = ((v >> i) & 1) ? '1' : '0';
    out[8] = '\0';
}

/* Convierte entero a binario de 48 bits */
void int_to_bin48(long long v, char *out) {
    for (int i = 47; i >= 0; i--)
        out[47-i] = ((v >> i) & 1) ? '1' : '0';
    out[48] = '\0';
}

/* Número de registro: "R0"→0, "R1"→1, ... */
int reg_num(const char *r) {
    if (r[0] == 'R' || r[0] == 'r')
        return atoi(r + 1);
    return 0;
}

/* Emitir un byte al buffer de salida */
void emitir_byte(const char *bits) {
    if (num_output >= MAX_OUTPUT) {
        fprintf(stderr, "Error: buffer de salida lleno\n");
        exit(1);
    }
    strncpy(output[num_output++], bits, 8);
    output[num_output-1][8] = '\0';
}

/* ════════════════════════════════════════
   TABLA DE ETIQUETAS
   ════════════════════════════════════════ */

void agregar_label(const char *nombre, int addr) {
    for (int i = 0; i < num_labels; i++) {
        if (strcmp(labels[i].nombre, nombre) == 0) {
            labels[i].addr = addr;
            return;
        }
    }
    strncpy(labels[num_labels].nombre, nombre, 63);
    labels[num_labels].addr = addr;
    num_labels++;
    fprintf(stderr, "[ENSAMBLADOR] Label '%s' → addr %d\n", nombre, addr);
}

int buscar_label(const char *nombre) {
    for (int i = 0; i < num_labels; i++)
        if (strcmp(labels[i].nombre, nombre) == 0)
            return labels[i].addr;
    fprintf(stderr, "[ENSAMBLADOR] Error: label '%s' no definida\n", nombre);
    return 0;
}

/* ════════════════════════════════════════
   EMISORES DE INSTRUCCIONES
   ════════════════════════════════════════ */

/* MOVI Ra, Imm8 — 4 bytes
   Formato: 1000000 [Ra:4b] [Imm8:8b] [pad:13b] */
void emit_movi(int ra, int imm8) {
    /* Formato: 1000000 [Ra:4b] [Imm8:8b] [pad:13b] = 32 bits
       Ejemplo MOVI R0, 5:
         1000000 0000 00000101 0000000000000
         Byte0=10000000 Byte1=00000000 Byte2=10100000 Byte3=00000000 */
    char total[33];
    char ra_bin[5];
    char imm8_bin[9];
    char pad[14];

    /* opcode fijo */
    strcpy(total, "1000000");

    /* Ra en 4 bits */
    for (int i = 3; i >= 0; i--)
        ra_bin[3-i] = ((ra >> i) & 1) ? '1' : '0';
    ra_bin[4] = '\0';
    strcat(total, ra_bin);

    /* Imm8 en 8 bits */
    for (int i = 7; i >= 0; i--)
        imm8_bin[7-i] = ((imm8 >> i) & 1) ? '1' : '0';
    imm8_bin[8] = '\0';
    strcat(total, imm8_bin);

    /* pad 13 bits */
    strcpy(pad, "0000000000000");
    strcat(total, pad);

    /* Emitir 4 bytes */
    char byte[9];
    for (int i = 0; i < 4; i++) {
        strncpy(byte, total + i*8, 8);
        byte[8] = '\0';
        emitir_byte(byte);
    }
}

/* ADDI Ra, Imm8 — 4 bytes
   Formato: 1000001 [Ra:4b] [Imm8:8b] [pad:13b] */
void emit_addi(int ra, int imm8) {
    /* Formato: 1000001 [Ra:4b] [Imm8:8b] [pad:13b] */
    char total[33];
    char ra_bin[5];
    char imm8_bin[9];

    strcpy(total, "1000001");

    for (int i = 3; i >= 0; i--)
        ra_bin[3-i] = ((ra >> i) & 1) ? '1' : '0';
    ra_bin[4] = '\0';
    strcat(total, ra_bin);

    for (int i = 7; i >= 0; i--)
        imm8_bin[7-i] = ((imm8 >> i) & 1) ? '1' : '0';
    imm8_bin[8] = '\0';
    strcat(total, imm8_bin);

    strcat(total, "0000000000000");

    char byte[9];
    for (int i = 0; i < 4; i++) {
        strncpy(byte, total + i*8, 8);
        byte[8] = '\0';
        emitir_byte(byte);
    }
}

/* Instrucciones RR — 2 bytes
   Formato: [opcode7] [Ra:4b] [Rb:4b] [pad:1b] */
void emit_rr(const char *opcode7, int ra, int rb) {
    char b0[9], b1[9];
    /* Byte 0: opcode7 + bit alto de Ra */
    unsigned int word = 0;
    /* Construir 16 bits: opcode7[6:0] Ra[3:0] Rb[3:0] pad[0] */
    unsigned int hi = 0;
    /* opcode7 es string de 7 chars '0'/'1' */
    for (int i = 0; i < 7; i++)
        hi = (hi << 1) | (opcode7[i] == '1' ? 1 : 0);
    unsigned int val16 = (hi << 9) | ((ra & 0xF) << 5) |
                         ((rb & 0xF) << 1) | 0;
    int_to_bin8((val16 >> 8) & 0xFF, b0);
    int_to_bin8((val16 >> 0) & 0xFF, b1);
    emitir_byte(b0);
    emitir_byte(b1);
}

/* Instrucciones R — 2 bytes (Rb=0, pad=0) */
void emit_r(const char *opcode7, int ra) {
    emit_rr(opcode7, ra, 0);
}

/* Instrucciones DIR — 8 bytes
   Formato: [opcode7] [Dir:48b] [pad:9b] */
void emit_dir(const char *opcode7, int dir) {
    char bits[65];
    unsigned int hi = 0;
    for (int i = 0; i < 7; i++)
        hi = (hi << 1) | (opcode7[i] == '1' ? 1 : 0);

    /* 64 bits: opcode7 + dir48 + pad9 */
    char b48[49];
    int_to_bin48((long long)dir, b48);

    /* Construir los 8 bytes */
    char full[65];
    full[0] = '\0';
    /* opcode7 */
    strncat(full, opcode7, 7);
    /* dir 48 bits */
    strncat(full, b48, 48);
    /* pad 9 bits */
    strncat(full, "000000000", 10);

    /* Emitir 8 bytes */
    char byte[9];
    for (int i = 0; i < 8; i++) {
        strncpy(byte, full + i*8, 8);
        byte[8] = '\0';
        emitir_byte(byte);
    }
}

/* Instrucciones R_DIR — 8 bytes
   Formato: [opcode7] [Ra:4b] [Dir:48b] [pad:5b] */
void emit_r_dir(const char *opcode7, int ra, int dir) {
    char b48[49];
    int_to_bin48((long long)dir, b48);

    char ra_bin[5];
    for (int i = 3; i >= 0; i--)
        ra_bin[3-i] = ((ra >> i) & 1) ? '1' : '0';
    ra_bin[4] = '\0';

    char full[65];
    full[0] = '\0';
    strncat(full, opcode7, 7);
    strncat(full, ra_bin, 4);
    strncat(full, b48, 48);
    strncat(full, "00000", 6);

    char byte[9];
    for (int i = 0; i < 8; i++) {
        strncpy(byte, full + i*8, 8);
        byte[8] = '\0';
        emitir_byte(byte);
    }
}

/* ════════════════════════════════════════
   ENSAMBLADO DE UNA LÍNEA
   ════════════════════════════════════════ */

void ensamblar_linea(const char *linea) {
    char buf[256];
    strncpy(buf, linea, 255);

    /* Quitar comentario # */
    char *com = strchr(buf, '#');
    if (com) *com = '\0';

    /* Quitar \r y \n */
    char *cr = strchr(buf, '\r');
    if (cr) *cr = '\0';
    char *nl = strchr(buf, '\n');
    if (nl) *nl = '\0';

    /* Saltar espacios iniciales */
    char *p = buf;
    while (*p == ' ' || *p == '\t') p++;

    /* Línea vacía */
    if (*p == '\0') return;

    /* Detectar etiqueta: termina en ':' */
    char *colon = strchr(p, ':');
    if (colon && (colon[1] == '\0' || colon[1] == ' ' ||
                  colon[1] == '\t' || colon[1] == '\r')) {
        *colon = '\0';
        if (pasada == 1)
            agregar_label(p, current_addr);
        return;
    }

    /* Leer mnemónico */
    char mnem[32];
    int i = 0;
    while (*p && *p != ' ' && *p != '\t' && i < 31)
        mnem[i++] = *p++;
    mnem[i] = '\0';

    /* Saltar espacios */
    while (*p == ' ' || *p == '\t') p++;

    /* ── Directivas de enlace ── */
    if (strcmp(mnem, "GLOBAL") == 0) {
        char nombre[64];
        sscanf(p, "%s", nombre);
        if (pasada == 2)
            printf("GLOBAL %s %d\n", nombre, current_addr);
        return;
    }
    if (strcmp(mnem, "EXTERN") == 0) {
        char nombre[64];
        sscanf(p, "%s", nombre);
        if (pasada == 2) {
            printf("EXTERN %s %d\n", nombre, current_addr);
            for (int j = 0; j < 8; j++)
                emitir_byte("00000000");
        }
        current_addr += 8;
        return;
    }

    /* ── Instrucciones de 1 byte ── */
    if (strcmp(mnem, "NOP")    == 0) {
        current_addr += 1;
        if (pasada == 2) emitir_byte("00000000");
        return;
    }
    if (strcmp(mnem, "HALT")   == 0) {
        current_addr += 1;
        if (pasada == 2) emitir_byte("00000001");
        return;
    }
    if (strcmp(mnem, "RET")    == 0) {
        current_addr += 1;
        if (pasada == 2) emitir_byte("00000010");
        return;
    }
    if (strcmp(mnem, "IRET")   == 0) {
        current_addr += 1;
        if (pasada == 2) emitir_byte("00000011");
        return;
    }
    if (strcmp(mnem, "FPU_OP") == 0) {
        current_addr += 1;
        if (pasada == 2) emitir_byte("00000100");
        return;

    if (!f2) {
        fprintf(stderr, "Error: no se puede abrir '%s'\n", argv[1]);
        return 1;
    }
    while (fgets(linea, sizeof(linea), f2))
        ensamblar_linea(linea);
    fclose(f2);

    /* ── SALIDA ── */
    FILE *out = stdout;
    if (argc == 3) {
        out = fopen(argv[2], "w");
        if (!out) {
            fprintf(stderr, "Error: no se puede crear '%s'\n", argv[2]);
            return 1;
        }
    }

    fprintf(stderr, "[ENSAMBLADOR] Bytes generados: %d\n", num_output);
    for (int i = 0; i < num_output; i++)
        fprintf(out, "%s\n", output[i]);

    if (argc == 3) fclose(out);
    return 0;
}

			case EOB_ACT_CONTINUE_SCAN:
				(yy_c_buf_p) =
					(yytext_ptr) + yy_amount_of_matched_text;

				yy_current_state = yy_get_previous_state(  );

				yy_cp = (yy_c_buf_p);
				yy_bp = (yytext_ptr) + YY_MORE_ADJ;
				goto yy_match;

			case EOB_ACT_LAST_MATCH:
				(yy_c_buf_p) =
				&YY_CURRENT_BUFFER_LVALUE->yy_ch_buf[(yy_n_chars)];

				yy_current_state = yy_get_previous_state(  );

				yy_cp = (yy_c_buf_p);
				yy_bp = (yytext_ptr) + YY_MORE_ADJ;
				goto yy_find_action;
			}
		break;
		}

	default:
		YY_FATAL_ERROR(
			"fatal flex scanner internal error--no action found" );
	} /* end of action switch */
		} /* end of scanning one token */
	} /* end of user's declarations */
} /* end of yylex */

/* yy_get_next_buffer - try to read in a new buffer
 *
 * Returns a code representing an action:
 *	EOB_ACT_LAST_MATCH -
 *	EOB_ACT_CONTINUE_SCAN - continue scanning from current position
 *	EOB_ACT_END_OF_FILE - end of file
 */
static int yy_get_next_buffer (void)
{
    	char *dest = YY_CURRENT_BUFFER_LVALUE->yy_ch_buf;
	char *source = (yytext_ptr);
	int number_to_move, i;
	int ret_val;

	if ( (yy_c_buf_p) > &YY_CURRENT_BUFFER_LVALUE->yy_ch_buf[(yy_n_chars) + 1] )
		YY_FATAL_ERROR(
		"fatal flex scanner internal error--end of buffer missed" );

	if ( YY_CURRENT_BUFFER_LVALUE->yy_fill_buffer == 0 )
		{ /* Don't try to fill the buffer, so this is an EOF. */
		if ( (yy_c_buf_p) - (yytext_ptr) - YY_MORE_ADJ == 1 )
			{
			/* We matched a single character, the EOB, so
			 * treat this as a final EOF.
			 */
			return EOB_ACT_END_OF_FILE;
			}

		else
			{
			/* We matched some text prior to the EOB, first
			 * process it.
			 */
			return EOB_ACT_LAST_MATCH;
			}
		}

	/* Try to read more data. */

	/* First move last chars to start of buffer. */
	number_to_move = (int) ((yy_c_buf_p) - (yytext_ptr) - 1);

	for ( i = 0; i < number_to_move; ++i )
		*(dest++) = *(source++);

	if ( YY_CURRENT_BUFFER_LVALUE->yy_buffer_status == YY_BUFFER_EOF_PENDING )
		/* don't do the read, it's not guaranteed to return an EOF,
		 * just force an EOF
		 */
		YY_CURRENT_BUFFER_LVALUE->yy_n_chars = (yy_n_chars) = 0;

	else
		{
			int num_to_read =
			YY_CURRENT_BUFFER_LVALUE->yy_buf_size - number_to_move - 1;

		while ( num_to_read <= 0 )
			{ /* Not enough room in the buffer - grow it. */

			/* just a shorter name for the current buffer */
			YY_BUFFER_STATE b = YY_CURRENT_BUFFER_LVALUE;

			int yy_c_buf_p_offset =
				(int) ((yy_c_buf_p) - b->yy_ch_buf);

			if ( b->yy_is_our_buffer )
				{
				int new_size = b->yy_buf_size * 2;

				if ( new_size <= 0 )
					b->yy_buf_size += b->yy_buf_size / 8;
				else
					b->yy_buf_size *= 2;

				b->yy_ch_buf = (char *)
					/* Include room in for 2 EOB chars. */
					yyrealloc( (void *) b->yy_ch_buf,
							 (yy_size_t) (b->yy_buf_size + 2)  );
				}
			else
				/* Can't grow it, we don't own it. */
				b->yy_ch_buf = NULL;

			if ( ! b->yy_ch_buf )
				YY_FATAL_ERROR( "out of dynamic memory in yy_create_buffer()" );

			(yy_c_buf_p) = &b->yy_ch_buf[yy_c_buf_p_offset];

			num_to_read = YY_CURRENT_BUFFER_LVALUE->yy_buf_size -
						number_to_move - 1;

			}

		if ( num_to_read > YY_READ_BUF_SIZE )
			num_to_read = YY_READ_BUF_SIZE;

		/* Read in more data. */
		YY_INPUT( (&YY_CURRENT_BUFFER_LVALUE->yy_ch_buf[number_to_move]),
			(yy_n_chars), num_to_read );

		YY_CURRENT_BUFFER_LVALUE->yy_n_chars = (yy_n_chars);
		}

	if ( (yy_n_chars) == 0 )
		{
		if ( number_to_move == YY_MORE_ADJ )
			{
			ret_val = EOB_ACT_END_OF_FILE;
			yyrestart( yyin  );
			}

		else
			{
			ret_val = EOB_ACT_LAST_MATCH;
			YY_CURRENT_BUFFER_LVALUE->yy_buffer_status =
				YY_BUFFER_EOF_PENDING;
			}
		}

	else
		ret_val = EOB_ACT_CONTINUE_SCAN;

	if (((yy_n_chars) + number_to_move) > YY_CURRENT_BUFFER_LVALUE->yy_buf_size) {
		/* Extend the array by 50%, plus the number we really need. */
		int new_size = (yy_n_chars) + number_to_move + ((yy_n_chars) >> 1);
		YY_CURRENT_BUFFER_LVALUE->yy_ch_buf = (char *) yyrealloc(
			(void *) YY_CURRENT_BUFFER_LVALUE->yy_ch_buf, (yy_size_t) new_size  );
		if ( ! YY_CURRENT_BUFFER_LVALUE->yy_ch_buf )
			YY_FATAL_ERROR( "out of dynamic memory in yy_get_next_buffer()" );
		/* "- 2" to take care of EOB's */
		YY_CURRENT_BUFFER_LVALUE->yy_buf_size = (int) (new_size - 2);
	}

	(yy_n_chars) += number_to_move;
	YY_CURRENT_BUFFER_LVALUE->yy_ch_buf[(yy_n_chars)] = YY_END_OF_BUFFER_CHAR;
	YY_CURRENT_BUFFER_LVALUE->yy_ch_buf[(yy_n_chars) + 1] = YY_END_OF_BUFFER_CHAR;

	(yytext_ptr) = &YY_CURRENT_BUFFER_LVALUE->yy_ch_buf[0];

	return ret_val;
}

/* yy_get_previous_state - get the state just before the EOB char was reached */

    static yy_state_type yy_get_previous_state (void)
{
	yy_state_type yy_current_state;
	char *yy_cp;
    
	yy_current_state = (yy_start);

	for ( yy_cp = (yytext_ptr) + YY_MORE_ADJ; yy_cp < (yy_c_buf_p); ++yy_cp )
		{
		YY_CHAR yy_c = (*yy_cp ? yy_ec[YY_SC_TO_UI(*yy_cp)] : 1);
		if ( yy_accept[yy_current_state] )
			{
			(yy_last_accepting_state) = yy_current_state;
			(yy_last_accepting_cpos) = yy_cp;
			}
		while ( yy_chk[yy_base[yy_current_state] + yy_c] != yy_current_state )
			{
			yy_current_state = (int) yy_def[yy_current_state];
			if ( yy_current_state >= 247 )
				yy_c = yy_meta[yy_c];
			}
		yy_current_state = yy_nxt[yy_base[yy_current_state] + yy_c];
		}

	return yy_current_state;
}

/* yy_try_NUL_trans - try to make a transition on the NUL character
 *
 * synopsis
 *	next_state = yy_try_NUL_trans( current_state );
 */
    static yy_state_type yy_try_NUL_trans  (yy_state_type yy_current_state )
{
	int yy_is_jam;
    	char *yy_cp = (yy_c_buf_p);

	YY_CHAR yy_c = 1;
	if ( yy_accept[yy_current_state] )
		{
		(yy_last_accepting_state) = yy_current_state;
		(yy_last_accepting_cpos) = yy_cp;
		}
	while ( yy_chk[yy_base[yy_current_state] + yy_c] != yy_current_state )
		{
		yy_current_state = (int) yy_def[yy_current_state];
		if ( yy_current_state >= 247 )
			yy_c = yy_meta[yy_c];
		}
	yy_current_state = yy_nxt[yy_base[yy_current_state] + yy_c];
	yy_is_jam = (yy_current_state == 246);

		return yy_is_jam ? 0 : yy_current_state;
}

#ifndef YY_NO_UNPUT

    static void yyunput (int c, char * yy_bp )
{
	char *yy_cp;
    
    yy_cp = (yy_c_buf_p);

	/* undo effects of setting up yytext */
	*yy_cp = (yy_hold_char);

	if ( yy_cp < YY_CURRENT_BUFFER_LVALUE->yy_ch_buf + 2 )
		{ /* need to shift things up to make room */
		/* +2 for EOB chars. */
		int number_to_move = (yy_n_chars) + 2;
		char *dest = &YY_CURRENT_BUFFER_LVALUE->yy_ch_buf[
					YY_CURRENT_BUFFER_LVALUE->yy_buf_size + 2];
		char *source =
				&YY_CURRENT_BUFFER_LVALUE->yy_ch_buf[number_to_move];

		while ( source > YY_CURRENT_BUFFER_LVALUE->yy_ch_buf )
			*--dest = *--source;

		yy_cp += (int) (dest - source);
		yy_bp += (int) (dest - source);
		YY_CURRENT_BUFFER_LVALUE->yy_n_chars =
			(yy_n_chars) = (int) YY_CURRENT_BUFFER_LVALUE->yy_buf_size;

		if ( yy_cp < YY_CURRENT_BUFFER_LVALUE->yy_ch_buf + 2 )
			YY_FATAL_ERROR( "flex scanner push-back overflow" );
		}

	*--yy_cp = (char) c;

	(yytext_ptr) = yy_bp;
	(yy_hold_char) = *yy_cp;
	(yy_c_buf_p) = yy_cp;
}

#endif

#ifndef YY_NO_INPUT
#ifdef __cplusplus
    static int yyinput (void)
#else
    static int input  (void)
#endif

{
	int c;
    
	*(yy_c_buf_p) = (yy_hold_char);

	if ( *(yy_c_buf_p) == YY_END_OF_BUFFER_CHAR )
		{
		/* yy_c_buf_p now points to the character we want to return.
		 * If this occurs *before* the EOB characters, then it's a
		 * valid NUL; if not, then we've hit the end of the buffer.
		 */
		if ( (yy_c_buf_p) < &YY_CURRENT_BUFFER_LVALUE->yy_ch_buf[(yy_n_chars)] )
			/* This was really a NUL. */
			*(yy_c_buf_p) = '\0';

		else
			{ /* need more input */
			int offset = (int) ((yy_c_buf_p) - (yytext_ptr));
			++(yy_c_buf_p);

			switch ( yy_get_next_buffer(  ) )
				{
				case EOB_ACT_LAST_MATCH:
					/* This happens because yy_g_n_b()
					 * sees that we've accumulated a
					 * token and flags that we need to
					 * try matching the token before
					 * proceeding.  But for input(),
					 * there's no matching to consider.
					 * So convert the EOB_ACT_LAST_MATCH
					 * to EOB_ACT_END_OF_FILE.
					 */

					/* Reset buffer status. */
					yyrestart( yyin );

					/*FALLTHROUGH*/

				case EOB_ACT_END_OF_FILE:
					{
					if ( yywrap(  ) )
						return 0;

					if ( ! (yy_did_buffer_switch_on_eof) )
						YY_NEW_FILE;
#ifdef __cplusplus
					return yyinput();
#else
					return input();
#endif
					}

				case EOB_ACT_CONTINUE_SCAN:
					(yy_c_buf_p) = (yytext_ptr) + offset;
					break;
				}
			}
		}

	c = *(unsigned char *) (yy_c_buf_p);	/* cast for 8-bit char's */
	*(yy_c_buf_p) = '\0';	/* preserve yytext */
	(yy_hold_char) = *++(yy_c_buf_p);

	return c;
}
#endif	/* ifndef YY_NO_INPUT */

/** Immediately switch to a different input stream.
 * @param input_file A readable stream.
 * 
 * @note This function does not reset the start condition to @c INITIAL .
 */
    void yyrestart  (FILE * input_file )
{
    
	if ( ! YY_CURRENT_BUFFER ){
        yyensure_buffer_stack ();
		YY_CURRENT_BUFFER_LVALUE =
            yy_create_buffer( yyin, YY_BUF_SIZE );
	}

	yy_init_buffer( YY_CURRENT_BUFFER, input_file );
	yy_load_buffer_state(  );
}

/** Switch to a different input buffer.
 * @param new_buffer The new input buffer.
 * 
 */
    void yy_switch_to_buffer  (YY_BUFFER_STATE  new_buffer )
{
    
	/* TODO. We should be able to replace this entire function body
	 * with
	 *		yypop_buffer_state();
	 *		yypush_buffer_state(new_buffer);
     */
	yyensure_buffer_stack ();
	if ( YY_CURRENT_BUFFER == new_buffer )
		return;

	if ( YY_CURRENT_BUFFER )
		{
		/* Flush out information for old buffer. */
		*(yy_c_buf_p) = (yy_hold_char);
		YY_CURRENT_BUFFER_LVALUE->yy_buf_pos = (yy_c_buf_p);
		YY_CURRENT_BUFFER_LVALUE->yy_n_chars = (yy_n_chars);
		}

	YY_CURRENT_BUFFER_LVALUE = new_buffer;
	yy_load_buffer_state(  );

	/* We don't actually know whether we did this switch during
	 * EOF (yywrap()) processing, but the only time this flag
	 * is looked at is after yywrap() is called, so it's safe
	 * to go ahead and always set it.
	 */
	(yy_did_buffer_switch_on_eof) = 1;
}

static void yy_load_buffer_state  (void)
{
    	(yy_n_chars) = YY_CURRENT_BUFFER_LVALUE->yy_n_chars;
	(yytext_ptr) = (yy_c_buf_p) = YY_CURRENT_BUFFER_LVALUE->yy_buf_pos;
	yyin = YY_CURRENT_BUFFER_LVALUE->yy_input_file;
	(yy_hold_char) = *(yy_c_buf_p);
}

/** Allocate and initialize an input buffer state.
 * @param file A readable stream.
 * @param size The character buffer size in bytes. When in doubt, use @c YY_BUF_SIZE.
 * 
 * @return the allocated buffer state.
 */
    YY_BUFFER_STATE yy_create_buffer  (FILE * file, int  size )
{
	YY_BUFFER_STATE b;
    
	b = (YY_BUFFER_STATE) yyalloc( sizeof( struct yy_buffer_state )  );
	if ( ! b )
		YY_FATAL_ERROR( "out of dynamic memory in yy_create_buffer()" );

	b->yy_buf_size = size;

	/* yy_ch_buf has to be 2 characters longer than the size given because
	 * we want to put in 2 end-of-buffer characters.
	 */
	b->yy_ch_buf = (char *) yyalloc( (yy_size_t) (b->yy_buf_size + 2)  );
	if ( ! b->yy_ch_buf )
		YY_FATAL_ERROR( "out of dynamic memory in yy_create_buffer()" );

	b->yy_is_our_buffer = 1;

	yy_init_buffer( b, file );

	return b;
}

/** Destroy the buffer.
 * @param b a buffer created with yy_create_buffer()
 * 
 */
    void yy_delete_buffer (YY_BUFFER_STATE  b )
{
    
	if ( ! b )
		return;

	if ( b == YY_CURRENT_BUFFER ) /* Not sure if we should pop here. */
		YY_CURRENT_BUFFER_LVALUE = NULL;

	if ( b->yy_is_our_buffer )
		yyfree( (void *) b->yy_ch_buf  );

	yyfree( (void *) b  );
}

/* Initializes or reinitializes a buffer.
 * This function is sometimes called more than once on the same buffer,
 * such as during a yyrestart() or at EOF.
 */
    static void yy_init_buffer  (YY_BUFFER_STATE  b, FILE * file )

{
	int oerrno = errno;
    
	yy_flush_buffer( b );

	b->yy_input_file = file;
	b->yy_fill_buffer = 1;

    /* If b is the current buffer, then yy_init_buffer was _probably_
     * called from yyrestart() or through yy_get_next_buffer.
     * In that case, we don't want to reset the lineno or column.
     */
    if (b != YY_CURRENT_BUFFER){
        b->yy_bs_lineno = 1;
        b->yy_bs_column = 0;
    }

        b->yy_is_interactive = file ? (isatty( fileno(file) ) > 0) : 0;
    
	errno = oerrno;
}

/** Discard all buffered characters. On the next scan, YY_INPUT will be called.
 * @param b the buffer state to be flushed, usually @c YY_CURRENT_BUFFER.
 * 
 */
    void yy_flush_buffer (YY_BUFFER_STATE  b )
{
    	if ( ! b )
		return;

	b->yy_n_chars = 0;

	/* We always need two end-of-buffer characters.  The first causes
	 * a transition to the end-of-buffer state.  The second causes
	 * a jam in that state.
	 */
	b->yy_ch_buf[0] = YY_END_OF_BUFFER_CHAR;
	b->yy_ch_buf[1] = YY_END_OF_BUFFER_CHAR;

	b->yy_buf_pos = &b->yy_ch_buf[0];

	b->yy_at_bol = 1;
	b->yy_buffer_status = YY_BUFFER_NEW;

	if ( b == YY_CURRENT_BUFFER )
		yy_load_buffer_state(  );
}

/** Pushes the new state onto the stack. The new state becomes
 *  the current state. This function will allocate the stack
 *  if necessary.
 *  @param new_buffer The new state.
 *  
 */
void yypush_buffer_state (YY_BUFFER_STATE new_buffer )
{
    	if (new_buffer == NULL)
		return;

	yyensure_buffer_stack();

	/* This block is copied from yy_switch_to_buffer. */
	if ( YY_CURRENT_BUFFER )
		{
		/* Flush out information for old buffer. */
		*(yy_c_buf_p) = (yy_hold_char);
		YY_CURRENT_BUFFER_LVALUE->yy_buf_pos = (yy_c_buf_p);
		YY_CURRENT_BUFFER_LVALUE->yy_n_chars = (yy_n_chars);
		}

	/* Only push if top exists. Otherwise, replace top. */
	if (YY_CURRENT_BUFFER)
		(yy_buffer_stack_top)++;
	YY_CURRENT_BUFFER_LVALUE = new_buffer;

	/* copied from yy_switch_to_buffer. */
	yy_load_buffer_state(  );
	(yy_did_buffer_switch_on_eof) = 1;
}

/** Removes and deletes the top of the stack, if present.
 *  The next element becomes the new top.
 *  
 */
void yypop_buffer_state (void)
{
    	if (!YY_CURRENT_BUFFER)
		return;

	yy_delete_buffer(YY_CURRENT_BUFFER );
	YY_CURRENT_BUFFER_LVALUE = NULL;
	if ((yy_buffer_stack_top) > 0)
		--(yy_buffer_stack_top);

	if (YY_CURRENT_BUFFER) {
		yy_load_buffer_state(  );
		(yy_did_buffer_switch_on_eof) = 1;
	}
}

/* Allocates the stack if it does not exist.
 *  Guarantees space for at least one push.
 */
static void yyensure_buffer_stack (void)
{
	yy_size_t num_to_alloc;
    
	if (!(yy_buffer_stack)) {

		/* First allocation is just for 2 elements, since we don't know if this
		 * scanner will even need a stack. We use 2 instead of 1 to avoid an
		 * immediate realloc on the next call.
         */
      num_to_alloc = 1; /* After all that talk, this was set to 1 anyways... */
		(yy_buffer_stack) = (struct yy_buffer_state**)yyalloc
								(num_to_alloc * sizeof(struct yy_buffer_state*)
								);
		if ( ! (yy_buffer_stack) )
			YY_FATAL_ERROR( "out of dynamic memory in yyensure_buffer_stack()" );

		memset((yy_buffer_stack), 0, num_to_alloc * sizeof(struct yy_buffer_state*));

		(yy_buffer_stack_max) = num_to_alloc;
		(yy_buffer_stack_top) = 0;
		return;
	}

	if ((yy_buffer_stack_top) >= ((yy_buffer_stack_max)) - 1){

		/* Increase the buffer to prepare for a possible push. */
		yy_size_t grow_size = 8 /* arbitrary grow size */;

		num_to_alloc = (yy_buffer_stack_max) + grow_size;
		(yy_buffer_stack) = (struct yy_buffer_state**)yyrealloc
								((yy_buffer_stack),
								num_to_alloc * sizeof(struct yy_buffer_state*)
								);
		if ( ! (yy_buffer_stack) )
			YY_FATAL_ERROR( "out of dynamic memory in yyensure_buffer_stack()" );

		/* zero only the new slots.*/
		memset((yy_buffer_stack) + (yy_buffer_stack_max), 0, grow_size * sizeof(struct yy_buffer_state*));
		(yy_buffer_stack_max) = num_to_alloc;
	}
}

/** Setup the input buffer state to scan directly from a user-specified character buffer.
 * @param base the character buffer
 * @param size the size in bytes of the character buffer
 * 
 * @return the newly allocated buffer state object.
 */
YY_BUFFER_STATE yy_scan_buffer  (char * base, yy_size_t  size )
{
	YY_BUFFER_STATE b;
    
	if ( size < 2 ||
	     base[size-2] != YY_END_OF_BUFFER_CHAR ||
	     base[size-1] != YY_END_OF_BUFFER_CHAR )
		/* They forgot to leave room for EOB's. */
		return NULL;

	b = (YY_BUFFER_STATE) yyalloc( sizeof( struct yy_buffer_state )  );
	if ( ! b )
		YY_FATAL_ERROR( "out of dynamic memory in yy_scan_buffer()" );

	b->yy_buf_size = (int) (size - 2);	/* "- 2" to take care of EOB's */
	b->yy_buf_pos = b->yy_ch_buf = base;
	b->yy_is_our_buffer = 0;
	b->yy_input_file = NULL;
	b->yy_n_chars = b->yy_buf_size;
	b->yy_is_interactive = 0;
	b->yy_at_bol = 1;
	b->yy_fill_buffer = 0;
	b->yy_buffer_status = YY_BUFFER_NEW;

	yy_switch_to_buffer( b  );

	return b;
}

/** Setup the input buffer state to scan a string. The next call to yylex() will
 * scan from a @e copy of @a str.
 * @param yystr a NUL-terminated string to scan
 * 
 * @return the newly allocated buffer state object.
 * @note If you want to scan bytes that may contain NUL values, then use
 *       yy_scan_bytes() instead.
 */
YY_BUFFER_STATE yy_scan_string (const char * yystr )
{
    
	return yy_scan_bytes( yystr, (int) strlen(yystr) );
}

/** Setup the input buffer state to scan the given bytes. The next call to yylex() will
 * scan from a @e copy of @a bytes.
 * @param yybytes the byte buffer to scan
 * @param _yybytes_len the number of bytes in the buffer pointed to by @a bytes.
 * 
 * @return the newly allocated buffer state object.
 */
YY_BUFFER_STATE yy_scan_bytes  (const char * yybytes, int  _yybytes_len )
{
	YY_BUFFER_STATE b;
	char *buf;
	yy_size_t n;
	int i;
    
	/* Get memory for full buffer, including space for trailing EOB's. */
	n = (yy_size_t) (_yybytes_len + 2);
	buf = (char *) yyalloc( n  );
	if ( ! buf )
		YY_FATAL_ERROR( "out of dynamic memory in yy_scan_bytes()" );

	for ( i = 0; i < _yybytes_len; ++i )
		buf[i] = yybytes[i];

	buf[_yybytes_len] = buf[_yybytes_len+1] = YY_END_OF_BUFFER_CHAR;

	b = yy_scan_buffer( buf, n );
	if ( ! b )
		YY_FATAL_ERROR( "bad buffer in yy_scan_bytes()" );

	/* It's okay to grow etc. this buffer, and we should throw it
	 * away when we're done.
	 */
	b->yy_is_our_buffer = 1;

	return b;
}

int yylex_destroy  (void)
{
    
    /* Pop the buffer stack, destroying each element. */
	while(YY_CURRENT_BUFFER){
		yy_delete_buffer( YY_CURRENT_BUFFER  );
		YY_CURRENT_BUFFER_LVALUE = NULL;
		yypop_buffer_state();
	}

	/* Destroy the stack itself. */
	yyfree((yy_buffer_stack) );
	(yy_buffer_stack) = NULL;

    /* Reset the globals. This function is important in a non-reentrant scanner so the next time
     * yylex() is called, initialization will occur. */
    yy_init_globals( );

    return 0;
}

%{
#include <stdio.h>
#include <stdlib.h>
#include <string.h>

#define MAX_MACROS   256
#define MAX_NOMBRE    64
#define MAX_VALOR    256
#define MAX_INCLUDES  32

typedef struct {
    char nombre[MAX_NOMBRE];
    char valor[MAX_VALOR];
} Macro;

Macro tabla_macros[MAX_MACROS];
int   num_macros = 0;

char archivos_incluidos[MAX_INCLUDES][256];
int  num_includes = 0;

"#define"[ \t]+[a-zA-Z_][a-zA-Z0-9_]*[ \t]+[^\n]* {
    char nombre[MAX_NOMBRE];
    char valor[MAX_VALOR];
    char *p = yytext + 7;
    while (*p == ' ' || *p == '\t') p++;
    int i = 0;
    while (*p && *p != ' ' && *p != '\t' && i < MAX_NOMBRE-1)
        nombre[i++] = *p++;
    nombre[i] = '\0';
    while (*p == ' ' || *p == '\t') p++;
    i = 0;
    while (*p && *p != '\n' && i < MAX_VALOR-1)
        valor[i++] = *p++;

}

void agregar_macro(const char *nombre, const char *valor) {
    for (int i = 0; i < num_macros; i++) {
        if (strcmp(tabla_macros[i].nombre, nombre) == 0) {
            strncpy(tabla_macros[i].valor, valor, MAX_VALOR-1);
            fprintf(stderr, "[PREPROCESADOR] Redefinida: %s = %s\n", nombre, valor);
            return;
        }
    }
    if (num_macros >= MAX_MACROS) {
        fprintf(stderr, "[PREPROCESADOR] Error: tabla llena\n");
        exit(1);
    }
    strncpy(tabla_macros[num_macros].nombre, nombre, MAX_NOMBRE-1);
    strncpy(tabla_macros[num_macros].valor,  valor,  MAX_VALOR-1);
    fprintf(stderr, "[PREPROCESADOR] Macro: %s = %s\n", nombre, valor);
    num_macros++;
}

const char *buscar_macro(const char *nombre) {
    for (int i = 0; i < num_macros; i++)
        if (strcmp(tabla_macros[i].nombre, nombre) == 0)
            return tabla_macros[i].valor;
    return NULL;
}

int ya_incluido(const char *filename) {
    for (int i = 0; i < num_includes; i++)
        if (strcmp(archivos_incluidos[i], filename) == 0)
            return 1;
    return 0;
}

void procesar_include(const char *filename) {
    if (ya_incluido(filename)) {
        fprintf(stderr, "[PREPROCESADOR] '%s' ya incluido\n", filename);
        return;
    }
    strncpy(archivos_incluidos[num_includes++], filename, 255);
    FILE *f = fopen(filename, "r");
    if (!f) {
        fprintf(stderr, "[PREPROCESADOR] Error: no se puede abrir '%s'\n", filename);
        exit(1);
    }
    fprintf(stderr, "[PREPROCESADOR] Incluyendo: %s\n", filename);
    int c;
    while ((c = fgetc(f)) != EOF)
        putchar(c);
    fclose(f);
}

int main(int argc, char *argv[]) {
    if (argc == 2) {
        yyin = fopen(argv[1], "r");
        if (!yyin) {
            fprintf(stderr, "Error: no se puede abrir '%s'\n", argv[1]);
            return 1;
        }
    }
    yylex();
    return 0;
}
